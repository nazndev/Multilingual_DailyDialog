# Cell 3: Run 1K subset pipeline
import subprocess

env = os.environ.copy()

# Step 01: Download
subprocess.run("python src/01_download.py", shell=True, cwd=REPO_DIR, env=env, check=True)

# Step 02: Preprocess

# Step 03: Translate (1k subset)
subprocess.run("python src/03_translate.py --config configs/translation_1k.yaml", shell=True, cwd=REPO_DIR, env=env, check=True)

# Step 05: Build SFT (writes to DATA_DIR/sft/multilingual_1k via translation_1k.yaml:sft_dir)

# Step 07: Eval (writes to REPORTS_DIR/eval_report_1k.md)
