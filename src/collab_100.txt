# Colab: 100-row demo end-to-end (bn by default)

# - configs/preprocess_100.yaml
# - configs/translation_100.yaml
# - configs/training_100.yaml
# - configs/eval_100.yaml

# Cell 1: Mount Drive + set persistent dirs
from google.colab import drive
drive.mount('/content/drive')

import os
COLAB_ROOT = "/content/drive/MyDrive/Multilingual_DailyDialog"
os.makedirs(COLAB_ROOT, exist_ok=True)

os.environ["PIP_CACHE_DIR"] = f"{COLAB_ROOT}/pip-cache"
os.environ["DATA_DIR"] = f"{COLAB_ROOT}/data"
os.environ["CACHE_DIR"] = f"{COLAB_ROOT}/cache"
os.environ["OUTPUTS_DIR"] = f"{COLAB_ROOT}/outputs"
os.environ["REPORTS_DIR"] = f"{COLAB_ROOT}/reports"
os.environ["TARGET_LANGS"] = "bn"


# Cell 2: Get code + install deps
REPO_DIR = f"{COLAB_ROOT}/repo"
if not os.path.exists(REPO_DIR):
    !git clone https://github.com/nazndev/Multilingual_DailyDialog.git "$REPO_DIR"
else:
    !git -C "$REPO_DIR" pull

%cd "$REPO_DIR"
!pip install -q -r requirements.txt


# Cell 3: Run 100-row demo pipeline
import subprocess

env = os.environ.copy()

subprocess.run("python src/01_download.py", shell=True, cwd=REPO_DIR, env=env, check=True)
subprocess.run("python src/02_preprocess.py --config configs/preprocess_100.yaml", shell=True, cwd=REPO_DIR, env=env, check=True)
subprocess.run("python src/03_translate.py --config configs/translation_100.yaml", shell=True, cwd=REPO_DIR, env=env, check=True)
subprocess.run("python src/05_build_sft.py --config configs/translation_100.yaml", shell=True, cwd=REPO_DIR, env=env, check=True)

# Fine-tune (very short)
subprocess.run("python src/06_train_sft.py --config configs/training_100.yaml", shell=True, cwd=REPO_DIR, env=env, check=True)

# Eval report includes: metrics + side-by-side examples (Zero-shot vs Fine-tuned)
subprocess.run("python src/07_eval.py --config configs/eval_100.yaml", shell=True, cwd=REPO_DIR, env=env, check=True)

print("Done. Key artifacts:")
print("- Processed 100:", os.path.join(env["DATA_DIR"], "processed_100"))
print("- Translated 100:", os.path.join(env["DATA_DIR"], "translated_dailydialog_en_bn_100"))
print("- SFT 100:", os.path.join(env["DATA_DIR"], "sft/multilingual_100"))
print("- Adapter:", os.path.join(env["OUTPUTS_DIR"], "model_100/lora_adapter"))
print("- Report:", os.path.join(env["REPORTS_DIR"], "eval_report_100.md"))
